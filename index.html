/* ============================================================ EXPORT DRAFT TO CSV (NEW) ============================================================ */ function exportDraft() { if (!state.picks.length) { notify("No picks to export"); return; } const sessionId = "S" + Date.now(); const headers = [ "pick_id","session_id","pick_number","round", "team_name","owner","player_name","tier" ]; const rows = state.picks.map((p, i) => { const round = Math.floor(i / state.teams.length) + 1; return [ "P" + (i + 1), sessionId, i + 1, round, `"${p.team}"`, `"${p.owner}"`, `"${p.player}"`, p.tier ]; }); const csvContent = [ headers.join(","), ...rows.map(r => r.join(",")) ].join("\n"); const blob = new Blob([csvContent], { type: "text/csv" }); const url = URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; a.download = `draft_${sessionId}.csv`; a.click(); URL.revokeObjectURL(url); notify("📊 Draft exported successfully!"); }
